    **3. `@Observable @MainActor final class CloudSyncStatusObserver`** with translator. Key decisions:

    - `private(set) var status: SyncStatus` (PATTERNS §2 line 171 — observer is the only writer; views read-only).
    - `private var lastSyncDate: Date?` — preserved across status changes so `.unavailable(lastSynced:)` carries the last-known-good date.
    - `@objc private func handleEvent(_:)` selector signature — the observer is registered via `addObserver(_:selector:name:object:)` (selector form, not block form). Use `selector: #selector(handleEvent(_:))`.
    - `@objc nonisolated private func handleEvent(_ notification: Notification)` declaration — the selector method must be reachable from the background queue Apple posts on; the body wraps work in `Task { @MainActor in self.applyEvent(...) }` to hop main (PATTERNS §2 line 192 — the recommended approach).
    - `private func applyEvent(...)` is `@MainActor` (default class isolation) and does the actual translation + status mutation.

    Verbatim skeleton:
    ```swift
    @Observable
    @MainActor
    final class CloudSyncStatusObserver {

        // MARK: - Logger (S2)
        private static let logger = Logger(
            subsystem: "com.lauterstar.gamekit",
            category: "cloudkit"
        )

        // MARK: - Published state (PATTERNS §2 line 171: private(set) — observer is only writer)
        private(set) var status: SyncStatus

        // MARK: - Private state
        private var lastSyncDate: Date?

        // MARK: - Init
        init(initialStatus: SyncStatus = .notSignedIn) {
            self.status = initialStatus
            // Selector form so we can reach a @objc method from background-queue
            // delivery; the @objc method then hops to MainActor.
            NotificationCenter.default.addObserver(
                self,
                selector: #selector(handleEvent(_:)),
                name: NSPersistentCloudKitContainer.eventChangedNotification,
                object: nil
            )
        }

        deinit {
            NotificationCenter.default.removeObserver(self)
        }

        // MARK: - Notification handler

        /// Called on whatever thread NotificationCenter dispatches; hops
        /// to MainActor before mutating state (T-06-state-bg-write mitigation).
        @objc nonisolated private func handleEvent(_ notification: Notification) {
            // Extract Event payload OFF main (no state mutation); then hop.
            // RESEARCH §A1: userInfo key verified.
            guard let event = notification.userInfo?[
                NSPersistentCloudKitContainer.eventNotificationUserInfoKey
            ] as? NSPersistentCloudKitContainer.Event else {
                return
            }

            let snapshot = (
                type: event.type,
                endDate: event.endDate,
                succeeded: event.succeeded,
                error: event.error
            )

            Task { @MainActor [weak self] in
                self?.applyEvent(
                    type: snapshot.type,
                    endDate: snapshot.endDate,
                    succeeded: snapshot.succeeded,
                    error: snapshot.error
                )
            }
        }

        /// Pure translator — single source of truth for the 4-state machine.
        /// Used by the production handleEvent path AND the #if DEBUG test seam
        /// because NSPersistentCloudKitContainer.Event has no public initializer
        /// (PATTERNS §S5).
        private func applyEvent(
            type: NSPersistentCloudKitContainer.EventType,
            endDate: Date?,
            succeeded: Bool,
            error: Error?
        ) {
            // RESEARCH §Pattern 5 lines 588-611:
            if endDate == nil {
                status = .syncing
                return
            }
            if succeeded {
                lastSyncDate = endDate
                status = .syncedAt(endDate ?? Date())
            } else {
                if let error {
                    Self.logger.error(
                        "CloudKit \(String(describing: type), privacy: .public) failed: \(error.localizedDescription, privacy: .public)"
                    )
                }
                status = .unavailable(lastSynced: lastSyncDate)
            }
        }

        // MARK: - DEBUG test seam (PATTERNS §S5)
        #if DEBUG
        /// Direct entry to the translator. Plan 06-02 tests use this because
        /// NSPersistentCloudKitContainer.Event has no public initializer.
        internal func applyEvent_forTesting(
            type: NSPersistentCloudKitContainer.EventType,
            endDate: Date?,
            succeeded: Bool,
            error: Error?
        ) {
            applyEvent(type: type, endDate: endDate, succeeded: succeeded, error: error)
        }
        #endif
    }
    ```

    **Critical implementation notes:**

    - The `@objc nonisolated handleEvent` body extracts the event snapshot OUTSIDE the `Task { @MainActor }` because reading `event.type/.endDate/.succeeded/.error` is thread-safe (Apple-provided `Event` value-type-like properties) but `status =` mutation MUST be on main.
