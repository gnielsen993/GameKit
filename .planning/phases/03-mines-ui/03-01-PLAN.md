    Sketch:
    ```swift
    import Foundation
    import SwiftUI
    #if canImport(UIKit)
    import UIKit
    #elseif canImport(AppKit)
    import AppKit
    #endif

    enum ColorVisionDeficiency: String, CaseIterable {
        case protanopia, deuteranopia, tritanopia
    }

    enum ColorVisionSimulator {
        // Brettel/Vienot/Mollon 1997 transform matrices for severe (100%) CVD,
        // applied in sRGB linear-RGB domain. References: Machado et al. 2009.
        private static let protanopiaMatrix = simd_double3x3(
            rows: [SIMD3(0.567, 0.433, 0.000),
                   SIMD3(0.558, 0.442, 0.000),
                   SIMD3(0.000, 0.242, 0.758)]
        )
        private static let deuteranopiaMatrix = simd_double3x3(
            rows: [SIMD3(0.625, 0.375, 0.000),
                   SIMD3(0.700, 0.300, 0.000),
                   SIMD3(0.000, 0.300, 0.700)]
        )
        private static let tritanopiaMatrix = simd_double3x3(
            rows: [SIMD3(0.950, 0.050, 0.000),
                   SIMD3(0.000, 0.433, 0.567),
                   SIMD3(0.000, 0.475, 0.525)]
        )

        static func simulate(_ srgb: SIMD3<Double>, cvd: ColorVisionDeficiency) -> SIMD3<Double> {
            let m: simd_double3x3
            switch cvd {
            case .protanopia:   m = protanopiaMatrix
            case .deuteranopia: m = deuteranopiaMatrix
            case .tritanopia:   m = tritanopiaMatrix
            }
            // Apply in linear-RGB. Apply gamma decode → matrix → gamma encode.
            let linear = srgbDecode(srgb)
            let transformed = m * linear
            return srgbEncode(transformed)
        }

        // ΔE2000 — sRGB → XYZ → CIELab → ΔE2000 distance.
        // Implementation: ~30-40 lines of standard color math.
        static func deltaE2000(_ a: SIMD3<Double>, _ b: SIMD3<Double>) -> Double {
            let labA = sRGBToLab(a)
            let labB = sRGBToLab(b)
            return cieDeltaE2000(labA, labB)
        }

        // ... helpers: srgbDecode, srgbEncode, sRGBToLab, cieDeltaE2000 ...
        // ... Color → SIMD3<Double> extraction (mirrors DesignKitTests.colorComponents(_:)) ...
    }
    ```

    //  ThemeGameNumberTests.swift
    //  DesignKitTests
    //
    //  Token contract for theme.gameNumber(_:) (D-13). Verifies:
    //    - Clamp to 1...8 (n=0 → palette[0]; n=9 → palette[7])
    //    - Every audit-set preset emits a length-8 palette via the resolver path
    //    - Resolver path (Theme.resolve(preset:scheme:)) is the unit under test —
    //      NOT raw PresetTheme declarations (RESEARCH Pitfall 6)

    import XCTest
    import SwiftUI
    @testable import DesignKit

    @MainActor
    final class ThemeGameNumberTests: XCTestCase {

        // Audit set per UI-SPEC §Theme Matrix
        private static let auditPresets: [ThemePreset] = [
            .forest, .bubblegum, .barbie, .cream, .dracula, .voltage
        ]

        func testEveryAuditPresetEmitsLength8Palette() {
            for preset in Self.auditPresets {
                for scheme in [ColorScheme.light, .dark] {
                    let theme = Theme.resolve(preset: preset, scheme: scheme)
                    let palette = theme.colors.gameNumberPaletteWongSafe ?? theme.colors.gameNumberPalette
                    XCTAssertEqual(palette.count, 8,
                        "\(preset) [\(scheme)] palette must be length 8 — got \(palette.count)")
                }
            }
        }

        func testGameNumberClampsBelowOne() {
            let theme = Theme.resolve(preset: .forest, scheme: .light)
            XCTAssertEqual(theme.gameNumber(0), theme.gameNumber(1),
                "n=0 must clamp to palette[0]")
            XCTAssertEqual(theme.gameNumber(-5), theme.gameNumber(1),
                "Negative n must clamp to palette[0]")
        }

        func testGameNumberClampsAboveEight() {
            let theme = Theme.resolve(preset: .forest, scheme: .light)
            XCTAssertEqual(theme.gameNumber(9), theme.gameNumber(8),
                "n=9 must clamp to palette[7]")
            XCTAssertEqual(theme.gameNumber(99), theme.gameNumber(8))
        }

        func testGameNumberReturnsExpectedClassicColors() {
            let theme = Theme.resolve(preset: .forest, scheme: .light)
            // Sanity: Classic palette is Minesweeper-traditional. n=1 must be a blue-ish color
            // (R < G < B). This is a structural sanity check, not a hex equality.
            guard let c1 = ColorVisionSimulator.sRGBComponents(of: theme.gameNumber(1)) else {
                return XCTFail("Cannot extract sRGB components — platform unsupported")
            }
            XCTAssertGreaterThan(c1.z, c1.x,
                "Classic n=1 should be predominantly blue (B > R)")
        }

        func testWongSafeOverrideTakesPrecedenceWhenSet() {
            // Construct a synthetic theme where wong-safe is set; verify gameNumber reads it
            let palette = (0..<8).map { Color(red: 0, green: 0, blue: Double($0) / 8.0) }
            let safe = (0..<8).map { Color(red: Double($0) / 8.0, green: 0, blue: 0) }
            let synthetic = ThemeColors(
                background: .white, surface: .white, surfaceElevated: .white, border: .black,
                textPrimary: .black, textSecondary: .gray, textTertiary: .gray,
                accentPrimary: .blue, accentSecondary: .purple, highlight: .yellow,
                success: .green, warning: .orange, danger: .red,
                fillPressed: .gray, fillSelected: .blue, fillDisabled: .gray,
                gameNumberPalette: palette,
                gameNumberPaletteWongSafe: safe
            )
            let charts = ChartTokens(/* …minimal placeholder values… */)
            let theme = Theme(colors: synthetic, charts: charts)
            // theme.gameNumber(1) should read from `safe`, NOT `palette`
            guard let got = ColorVisionSimulator.sRGBComponents(of: theme.gameNumber(1)) else {
                return XCTFail("Cannot extract sRGB components")
            }
            XCTAssertGreaterThan(got.x, 0.0,  "Override path should produce non-zero red channel")
            XCTAssertEqual(got.y, 0.0, accuracy: 0.01)
            XCTAssertEqual(got.z, 0.0, accuracy: 0.01)
        }
    }

    //  - Loud presets (voltage / barbie) MAY ship `gameNumberPaletteWongSafe`
    //    override; the resolver-applied palette is the unit under test.
    //
    //  Reference: Machado et al. 2009, Brettel/Vienot/Mollon 1997.
    //  Threshold: ΔE2000 ≥ 10 between adjacent palette entries — chosen to
    //  match Wong-palette guidance (Nature Methods 2011); below 10 = "easily
    //  confusable" under CVD.
    //

    import XCTest
    import SwiftUI
    @testable import DesignKit

    @MainActor
    final class GameNumberPaletteWongTests: XCTestCase {

        private static let deltaThreshold: Double = 10.0
        private static let auditPresets: [ThemePreset] = [
            .forest, .bubblegum, .barbie, .cream, .dracula, .voltage
        ]

        /// Classic forest is the default first-launch preset. Per D-15 it MUST
        /// pass the Wong audit unconditionally.
        func testForestPalettePassesAllThreeCVDsUnconditionally() {
            let theme = Theme.resolve(preset: .forest, scheme: .light)
            let palette = theme.colors.gameNumberPaletteWongSafe ?? theme.colors.gameNumberPalette
            XCTAssertEqual(palette.count, 8)

            for cvd in ColorVisionDeficiency.allCases {
                let failures = adjacentPairFailures(palette: palette, cvd: cvd)
                XCTAssertTrue(failures.isEmpty,
                    "Forest [\(cvd)] adjacent-pair failures: \(failures)")
            }
        }

        /// Every audit preset's resolver-emitted palette (default OR override)
        /// must satisfy ΔE ≥ 10. If the default fails AND no override ships,
        /// this surfaces the gap so the executor adds an override.
        func testEveryAuditPresetResolverOutputIsWongSafe() {
            for preset in Self.auditPresets {
                let theme = Theme.resolve(preset: preset, scheme: preferredScheme(for: preset))
                let palette = theme.colors.gameNumberPaletteWongSafe ?? theme.colors.gameNumberPalette
                XCTAssertEqual(palette.count, 8, "\(preset) palette length")

                for cvd in ColorVisionDeficiency.allCases {
                    let failures = adjacentPairFailures(palette: palette, cvd: cvd)
                    XCTAssertTrue(failures.isEmpty,
                        "\(preset) [\(cvd)] adjacent-pair failures: \(failures) — ship `gameNumberPaletteWongSafe` override (D-15)")
                }
            }
        }

        // MARK: - Helpers

        private func adjacentPairFailures(palette: [Color], cvd: ColorVisionDeficiency)
            -> [(Int, Int, Double)] {
            var fails: [(Int, Int, Double)] = []
            for i in 0..<(palette.count - 1) {
                guard let a = ColorVisionSimulator.sRGBComponents(of: palette[i]),
                      let b = ColorVisionSimulator.sRGBComponents(of: palette[i + 1])
                else { continue }
                let aSim = ColorVisionSimulator.simulate(a, cvd: cvd)
                let bSim = ColorVisionSimulator.simulate(b, cvd: cvd)
                let dE = ColorVisionSimulator.deltaE2000(aSim, bSim)
                if dE < Self.deltaThreshold {
                    fails.append((i + 1, i + 2, dE))
                }
            }
            return fails
        }

        private func preferredScheme(for preset: ThemePreset) -> ColorScheme {
            // Loud / Moody presets prefer dark; Sweet / Soft / Bright prefer light;
            // Classic supports both — pick light for parity. Adjust if a preset's
            // light/dark palettes differ structurally.
            switch preset {
            case .dracula, .voltage: return .dark
            default:                  return .light
            }
        }
    }
