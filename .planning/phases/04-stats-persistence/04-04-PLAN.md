    ```swift
    import SwiftUI

    @Observable
    @MainActor
    final class SettingsStore {

        // MARK: - Stored flags

        /// P6 PERSIST-04 reads this. P4 default is false; the .private(...)
        /// branch in GameKitApp.init never executes in production yet, but

        var cloudSyncEnabled: Bool {

                userDefaults.set(cloudSyncEnabled, forKey: Self.cloudSyncEnabledKey)
            }
        }

        // MARK: - Private

        private let userDefaults: UserDefaults

        // MARK: - Constants

        /// UserDefaults key for cloud-sync flag (D-28).
        /// Renaming = preference loss for any user who already toggled the flag.
        static let cloudSyncEnabledKey = "gamekit.cloudSyncEnabled"

        // MARK: - Init

        init(userDefaults: UserDefaults = .standard) {
            self.userDefaults = userDefaults
            self.cloudSyncEnabled = userDefaults.bool(forKey: Self.cloudSyncEnabledKey)
        }
    }

    // MARK: - EnvironmentKey injection (RESEARCH §Pattern 5)

    private struct SettingsStoreKey: EnvironmentKey {
        @MainActor static let defaultValue = SettingsStore()
    }

    extension EnvironmentValues {
        var settingsStore: SettingsStore {
            get { self[SettingsStoreKey.self] }
            set { self[SettingsStoreKey.self] = newValue }
        }
    }
    ```

    **Why `final class` (not `actor`):** Swift Observation `@Observable` macro generates SwiftUI-compatible change tracking on classes; `actor` types are not directly observable. `@MainActor` annotation provides the actor isolation boundary equivalent.

    **Why `userDefaults.bool(forKey:)` returns `false` for missing keys** (matches D-28 default): UserDefaults' `bool(forKey:)` returns `false` for unset keys per Apple docs. No explicit default registration needed for v1; if future flags need a non-`false` default, register them via `userDefaults.register(defaults:)` in `init`.

    **`@MainActor` on `defaultValue`:** SwiftUI's `EnvironmentKey.defaultValue` is normally `nonisolated`; with the `@MainActor` annotation on the class, the default value's construction must also occur on main. Swift 6 compiler accepts the `@MainActor static let` form per RESEARCH §Pattern 5 example. If the build fails with isolation diagnostic, fall back to making the static a `nonisolated(unsafe)` constant — but try the annotated form first.

    **No `import SwiftData`** — SettingsStore has no model context; it's just a UserDefaults shim. The container construction lives in `GameKitApp.swift` (Task 2).
