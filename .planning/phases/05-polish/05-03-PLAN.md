    ```swift
    //
    //  Haptics.swift
    //  gamekit
    //
    //  P5 (D-07/D-10/D-11): @MainActor enum wrapping CHHapticEngine for AHAP file
    //  playback. Single shared engine, lazy-loaded on first playAHAP call.
    //  Failure is non-fatal: log via os.Logger and silent no-op.
    //  Gated at the source on hapticsEnabled (D-10) — no view-layer plumbing of the toggle.
    //

    import Foundation
    import CoreHaptics
    import os

    @MainActor
    enum Haptics {
        private static var engine: CHHapticEngine?
        private static let logger = Logger(
            subsystem: "com.lauterstar.gamekit",
            category: "haptics"
        )

        /// Play a bundled `.ahap` haptic pattern. Gated at the source on
        /// `hapticsEnabled` (per D-10). Failure is non-fatal — logs and returns.
        ///
        /// - Parameters:
        ///   - name: the AHAP file basename (without extension), e.g. "win" / "loss"
        ///   - hapticsEnabled: pass `settingsStore.hapticsEnabled` from the call site
        static func playAHAP(named name: String, hapticsEnabled: Bool) {
            guard hapticsEnabled else { return }                    // D-10 gate
            guard CHHapticEngine.capabilitiesForHardware().supportsHaptics else { return }
            guard let url = Bundle.main.url(forResource: name, withExtension: "ahap") else {
                logger.error("AHAP not found in bundle: \(name, privacy: .public)")
                return
            }
            do {
                try ensureEngine()
                let pattern = try CHHapticPattern(contentsOf: url)
                let player = try engine?.makePlayer(with: pattern)
                try player?.start(atTime: 0)
            } catch {
                logger.error(
                    "AHAP playback failed for \(name, privacy: .public): \(error.localizedDescription, privacy: .public)"
                )
                // Non-fatal: drop the engine so next call re-initializes
                engine = nil
            }
        }

        private static func ensureEngine() throws {
            if engine != nil { return }
            let newEngine = try CHHapticEngine()
            newEngine.resetHandler = { Self.engine = nil }
            newEngine.stoppedHandler = { _ in Self.engine = nil }
            try newEngine.start()
            engine = newEngine
        }

        // MARK: - Test seam (do NOT call from production)
        #if DEBUG
        internal static var hasInitializedEngineForTesting: Bool { engine != nil }
        #endif
    }
    ```

    Key design choices (all locked by CONTEXT/PATTERNS):
    - **`@MainActor enum Haptics`** with static methods (D-11 — no instance).
    - **Gating at the source** (D-10) — `hapticsEnabled` is the FIRST guard; engine is never touched if disabled.
    - **Hardware capability check** — silently no-op on devices without haptic support (iPad-ish, accessibility iOS Settings haptics-off — though latter actually still runs the call; this guards on hardware only).
    - **Lazy engine + reset/stoppedHandler** (D-11) — engine cleared on system reset so next call re-initializes.
    - **Non-fatal failure** (D-11) — log via `os.Logger(subsystem: "com.lauterstar.gamekit", category: "haptics")` matching GameStats.swift:47-50 + StatsExporter:39 pattern; `privacy: .public` on error description; reset engine on catch.
    - **`#if DEBUG`-gated test seam** — keeps the engine-initialization test feasible without polluting production API. The `internal` accessor is only visible via `@testable import gamekit`.

    ```swift
    //
    //  SFXPlayer.swift
    //  gamekit
    //
    //  P5 (D-08/D-09/D-10/D-12): @MainActor final class wrapping AVAudioPlayer
    //  for tap/win/loss CAF playback. Three preloaded players, AVAudioSession.ambient
    //  category (does not duck user music). Gated at the source on sfxEnabled (D-10).
    //  Constructed at GameKitApp.init() AFTER SettingsStore; injected via custom EnvironmentKey
    //  mirroring P4 SettingsStore D-29 pattern.
    //

    import Foundation
    import AVFoundation
    import SwiftUI
    import os

    enum SFXEvent: Sendable {
        case tap
        case win
        case loss
    }

    @MainActor
    final class SFXPlayer {

        private let tapPlayer: AVAudioPlayer?
        private let winPlayer: AVAudioPlayer?
        private let lossPlayer: AVAudioPlayer?
        private let logger = Logger(
            subsystem: "com.lauterstar.gamekit",
            category: "audio"
        )

        init() {
            // D-09: AVAudioSession.ambient — does not duck user music
            do {
                try AVAudioSession.sharedInstance().setCategory(.ambient, mode: .default)
            } catch {
                // Logger is not yet `self.logger` — use Logger directly here pre-self
                Logger(subsystem: "com.lauterstar.gamekit", category: "audio")
                    .error("Failed to set AVAudioSession .ambient: \(error.localizedDescription, privacy: .public)")
            }

            // D-08: preload all three players via prepareToPlay()
            self.tapPlayer = Self.makePlayer(name: "tap")
            self.winPlayer = Self.makePlayer(name: "win")
            self.lossPlayer = Self.makePlayer(name: "loss")
        }

        private static func makePlayer(name: String) -> AVAudioPlayer? {
            guard let url = Bundle.main.url(forResource: name, withExtension: "caf") else {
                Logger(subsystem: "com.lauterstar.gamekit", category: "audio")
                    .error("CAF not found in bundle: \(name, privacy: .public)")
                return nil
            }
            do {
                let player = try AVAudioPlayer(contentsOf: url)
                player.prepareToPlay()
                return player
            } catch {
                Logger(subsystem: "com.lauterstar.gamekit", category: "audio")
                    .error("AVAudioPlayer failed for \(name, privacy: .public): \(error.localizedDescription, privacy: .public)")
                return nil
            }
        }

        /// Play a preloaded SFX cue. Gated at the source on `sfxEnabled` (D-10).
        func play(_ event: SFXEvent, sfxEnabled: Bool) {
            #if DEBUG
            lastInvocationAttempt = (event, sfxEnabled)
            #endif
            guard sfxEnabled else { return }              // D-10 gate
            #if DEBUG
            lastPlayedEvent = event
            #endif
            switch event {
            case .tap:  tapPlayer?.play()
            case .win:  winPlayer?.play()
            case .loss: lossPlayer?.play()
            }
        }

        // MARK: - Test seam (do NOT call from production)
        #if DEBUG
        internal var lastInvocationAttempt: (event: SFXEvent, enabled: Bool)?
        internal var lastPlayedEvent: SFXEvent?
        internal var preloadedTap: AVAudioPlayer? { tapPlayer }
        internal var preloadedWin: AVAudioPlayer? { winPlayer }
        internal var preloadedLoss: AVAudioPlayer? { lossPlayer }
        #endif
    }

    // MARK: - EnvironmentKey injection (mirror SettingsStore.swift:67-78)

    private struct SFXPlayerKey: EnvironmentKey {
        @MainActor static let defaultValue = SFXPlayer()
    }

    extension EnvironmentValues {
        var sfxPlayer: SFXPlayer {
            get { self[SFXPlayerKey.self] }
            set { self[SFXPlayerKey.self] = newValue }
        }
    }
    ```

    Note: `import SwiftUI` is required because `EnvironmentKey` and `EnvironmentValues` live there. This is acceptable — SettingsStore.swift also imports SwiftUI for the same reason. Core/Haptics.swift does NOT need SwiftUI because its public API is `static func` only (no env-key injection — view sites import Core/Haptics directly).
