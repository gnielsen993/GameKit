```swift
@Observable
@MainActor
final class SettingsStore {
    var cloudSyncEnabled: Bool {
        didSet {
            userDefaults.set(cloudSyncEnabled, forKey: Self.cloudSyncEnabledKey)
        }
    }
    private let userDefaults: UserDefaults
    static let cloudSyncEnabledKey = "gamekit.cloudSyncEnabled"

    init(userDefaults: UserDefaults = .standard) {
        self.userDefaults = userDefaults
        self.cloudSyncEnabled = userDefaults.bool(forKey: Self.cloudSyncEnabledKey)
    }
}

private struct SettingsStoreKey: EnvironmentKey {
    @MainActor static let defaultValue = SettingsStore()
}

extension EnvironmentValues {
    var settingsStore: SettingsStore {
        get { self[SettingsStoreKey.self] }
        set { self[SettingsStoreKey.self] = newValue }
    }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| view↔store | UserDefaults is read by VM/views with no cryptographic protection — appropriate for non-secret preference flags |
| store↔disk | UserDefaults persists to plist; iOS sandbox isolates per-app |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | UserDefaults flag values (e.g. user toggling hasSeenIntro back to false via debug command) | accept | These are user preferences with no security boundary; tampering yields no privilege escalation. Documented in CONTEXT D-23 — the flag is a UX hint, not a guard. |
| T-05-02 | Information Disclosure | Settings flags persisted in plain plist | accept | No PII or secret data; flags are 3 booleans (audio/haptic prefs + onboarding completion). Matches existing `cloudSyncEnabled` posture. |
| T-05-03 | Denial of Service | Malformed UserDefaults entry corrupting init | mitigate | `object(forKey:) as? Bool ?? true` defensive cast survives type mismatch; `bool(forKey:)` returns `false` for missing/wrong-type keys (Apple-documented behavior). Tested in `unsetHapticsEnabledKey_returnsTrueByDefault`. |
</threat_model>
