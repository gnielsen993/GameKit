            .background(

                        : Color.clear
                    )
                )
                .contentShape(Capsule())
            }
            .buttonStyle(.plain)
            .accessibilityLabel(Text(label))
            .accessibilityAddTraits(isActive ? [.isButton, .isSelected] : .isButton)
        }
    }
