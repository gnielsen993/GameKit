    ```swift
    //  Animation gating: .transition collapses to .identity when reduceMotion
    //  OR !animationsEnabled (per CONTEXT D-13-ANIM; v1.0 05-06 D-04 contract).
    //  Haptics fire via VideoModeBanner's .sensoryFeedback (FIRST-guarded per
    //  CONTEXT D-13-HAPTICS in VideoModeBanner.swift).

    import SwiftUI
    import DesignKit

    extension MinesweeperGameView {

        /// Video Mode end-state banner — pill at the edge opposite the picked
        /// PiP zone per VideoModeBannerRouter / 08-BANNER-PLACEMENT.md.
        /// Replaces the full-screen endStateOverlay on the Video Mode path
        /// only. Renders inside the same ZStack the win-sweep + confetti use,
        /// gated by `endCardVisible` so the celebration pre-roll lands first
        /// (CONTEXT D-07 — celebration unchanged).
        @ViewBuilder
        func videoModeEndBanner(outcome: GameOutcome) -> some View {
            let anchor = VideoModeBannerRouter.anchor(for: videoModeStore.location)
            let content = bannerContent(for: outcome)

            VStack(spacing: 0) {
                if anchor.edge == .bottom {
                    Spacer(minLength: 0)
                }
                VideoModeBanner(
                    theme: theme,
                    content: content,
                    location: videoModeStore.location,
                    hapticsEnabled: settingsStore.hapticsEnabled,
                    reduceMotion: reduceMotion,
                    animationsEnabled: settingsStore.animationsEnabled
                )
                .padding(.vertical, theme.spacing.m)
                if anchor.edge == .top {
                    Spacer(minLength: 0)
                }
            }
            .transition((reduceMotion || !settingsStore.animationsEnabled) ? .identity : .opacity)
        }

        /// Build the per-outcome VideoModeBannerContent for Minesweeper.
        /// Copy + a11y label mirror MinesweeperEndStateCard.swift:82-115 verbatim
        /// per CONTEXT D-01 (D-13-COPY LOCKED).
        private func bannerContent(for outcome: GameOutcome) -> VideoModeBannerContent {
            switch outcome {
            case .win:
                return VideoModeBannerContent(
                    outcome: .win,
                    title: String(localized: "You won!"),
                    subtitle: nil,
                    primaryButtonLabel: String(localized: "Restart"),
                    onPrimary: { viewModel.restart() },
                    accessibilityLabel: "You won! Time: \(formatBannerElapsed(viewModel.frozenElapsed))"
                )
            case .loss:
                let a11y: String
                if let ctx = viewModel.lossContext {
                    a11y = "Bad luck. \(ctx.minesHit) mines hit, \(ctx.safeCellsRemaining) safe cells left."
                } else {
                    a11y = "Bad luck."
                }
                return VideoModeBannerContent(
                    outcome: .loss,
                    title: String(localized: "Bad luck"),
                    subtitle: nil,
                    primaryButtonLabel: String(localized: "Restart"),
                    onPrimary: { viewModel.restart() },
                    accessibilityLabel: a11y
                )
            }
        }

        /// Duplicate of the elapsed formatter at MinesweeperEndStateCard.swift:91-100.
        /// Per CLAUDE.md §4 a 2-call-site duplicate within one game is below the
        /// DesignKit-promotion bar; the existing EndStateCard keeps its private
        /// formatBranchELapsed verbatim per D-13-OFFPATH.
        private func formatBannerElapsed(_ t: TimeInterval) -> String {
            let total = max(0, Int(t))
            let h = total / 3600
            let m = (total % 3600) / 60
            let s = total % 60
            if h > 0 { return String(format: "%d:%02d:%02d", h, m, s) }
            return String(format: "%d:%02d", m, s)
        }
    }
    ```
