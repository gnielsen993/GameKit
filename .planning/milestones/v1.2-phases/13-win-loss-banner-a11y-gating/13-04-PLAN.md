    ```swift
    //
    //  NonogramGameView+EndBanner.swift
    //  gamekit
    //
    //  Phase 13 — Video Mode win/loss banner adoption for Nonogram.
    //  Mirrors MinesweeperGameView+EndBanner.swift (Plan 13-02) and
    //  MergeGameView+EndBanner.swift (Plan 13-03) — sibling pattern keeps
    //  banner content factories close to the per-game adopter.
    //
    //  Off-path (videoModeStore.isEnabled == false) keeps the existing
    //  NonogramEndStateCard byte-identical per CONTEXT D-13-OFFPATH + VIDEO-13.
    //
    //  Copy + a11y mirror NonogramEndStateCard verbatim per CONTEXT D-01
    //  (D-13-COPY LOCKED): "Solved" headline + puzzle title subtitle on .won;
    //  "Out of lives" headline on .gameOver. CTA: "New puzzle" / "Try again".
    //
    //  Single-action lock per CONTEXT D-02 + 08-BANNER-PLACEMENT D-11: the
    //  existing NonogramEndStateCard "Change size" secondary action is DROPPED
    //  — size selection reachable from the NonogramToolbarMenu.
    //

    import SwiftUI
    import DesignKit

    extension NonogramGameView {

        /// Video Mode end-state banner — pill at the edge opposite the picked
        /// PiP zone per VideoModeBannerRouter / 08-BANNER-PLACEMENT.md.
        /// Replaces the full-screen endStateOverlay on the Video Mode path
        /// only.
        @ViewBuilder
        func videoModeEndBanner() -> some View {
            let anchor = VideoModeBannerRouter.anchor(for: videoModeStore.location)
            let content = bannerContent()

            VStack(spacing: 0) {
                if anchor.edge == .bottom {
                    Spacer(minLength: 0)
                }
                VideoModeBanner(
                    theme: theme,
                    content: content,
                    location: videoModeStore.location,
                    hapticsEnabled: settingsStore.hapticsEnabled,
                    reduceMotion: reduceMotion,
                    animationsEnabled: settingsStore.animationsEnabled
                )
                .padding(.vertical, theme.spacing.m)
                if anchor.edge == .top {
                    Spacer(minLength: 0)
                }
            }
            .transition((reduceMotion || !settingsStore.animationsEnabled) ? .identity : .opacity)
        }

        /// Build the per-state VideoModeBannerContent for Nonogram.
        /// Reads viewModel.state directly (mirrors NonogramGameView.swift:164-190
        /// endStateOverlay shape — no parameter; same source-of-truth state).
        ///
        /// Copy + a11y mirror NonogramEndStateCard.swift:77-85 verbatim per
        /// CONTEXT D-01 (D-13-COPY LOCKED) and UI-SPEC §Accessibility Contract.
        private func bannerContent() -> VideoModeBannerContent {
            if viewModel.state == .won {
                let puzzleTitle = viewModel.currentPuzzle?.title ?? ""
                let subtitle: String? = puzzleTitle.isEmpty ? nil : puzzleTitle
                let a11y: String
                if !puzzleTitle.isEmpty {
                    a11y = "Solved. \(puzzleTitle). Time: \(formatBannerElapsed(viewModel.frozenElapsed))"
                } else {
                    a11y = "Solved. Time: \(formatBannerElapsed(viewModel.frozenElapsed))"
                }
                return VideoModeBannerContent(
                    outcome: .win,
                    title: String(localized: "Solved"),
                    subtitle: subtitle,
                    primaryButtonLabel: String(localized: "New puzzle"),
                    onPrimary: { viewModel.newPuzzle() },
                    accessibilityLabel: a11y
                )
            } else {
                // .gameOver (Lives 3-strikes) — and any non-.won terminal
                // state. Per NonogramGameView.swift:173 the off-path renders
                // .gameOver for any non-.won terminal, so the banner mirrors.
                return VideoModeBannerContent(
                    outcome: .loss,
                    title: String(localized: "Out of lives"),
                    subtitle: nil,
                    primaryButtonLabel: String(localized: "Try again"),
                    onPrimary: { viewModel.restart() },
                    accessibilityLabel: "Out of lives. You used all 3 lives."
                )
            }
        }

        /// Duplicate of NonogramEndStateCard.swift:91-100 formatter. Per CLAUDE.md §4
        /// a 2-call-site duplicate within one game is below the DesignKit-promotion
        /// bar; the existing card keeps its private formatter verbatim per D-13-OFFPATH.
        private func formatBannerElapsed(_ t: TimeInterval) -> String {
            let total = max(0, Int(t))
            let h = total / 3600
            let m = (total % 3600) / 60
            let s = total % 60
            if h > 0 { return String(format: "%d:%02d:%02d", h, m, s) }
            return String(format: "%d:%02d", m, s)
        }
    }
    ```
